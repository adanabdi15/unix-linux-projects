#!/bin/bash
echo "#####Welcome to my QUIZ####"
echo -e "Type your name-->\c"
read QUIZOWNER
touch $QUIZOWNER.txt
echo "Begin your quiz $QUIZOWNER..."

###########################################

echo -e "No 1: The sky is __________
1. Green \n2. Blue \n3. Yellow \n4. Black 
Enter answer--->\c"
read ANSWER
	if [ $ANSWER = 2 ]
	then
	echo "Correct!"
	Question1=1
	echo "Question 1: $Question1" >> $QUIZOWNER.txt

	else 
	echo "Incorrect! Try again."
	Question1=0

	fi

echo -e "\nNo 2: 4 + 2 = ______
1.6 \n2. 2 \n3. 12 \n4. 21
Enter answer---->\c"
read ANSWER
	if [ $ANSWER = 1 ]
	then
	echo "Correct!"
	Question2=1
	echo "Question 2: $Question2" >> $QUIZOWNER.txt
	
	else
	echo "Incorrect! Try again."
	Question2=0

	fi


echo -e "\nNo 3: Cars have _______ wheels?
1.3 \n2. 6\n3. 4 \n4. 7
Enter answer--->\c"
	read ANSWER

	if [ $ANSWER = 3 ]
	then
	echo "Correct!"
	Question3=1
	echo "Question 3: $Question3" >> $QuIZOWNER.txt

	else 
	echo "Incorrect! Try again."
	Question3=0

	fi

Total=$(( $Question1+$Question2+$Question3 ))
echo "You Scored $Total points!"

cat $QUIZOWNER.txt
